28 /*
29  * Versioned structure macros, used to access FF7 and FF8 data in a portable
30  * way. Allows massive, relatively straightforward code reuse between the two
31  * games. Watch out for block statements and using expressions with side
32  * effects! These macros can have unintended effects if you're not careful.
33  */
34 
35 /*
36  * VOBJ - Define a versioned object, structure must have FF7 and FF8 variants
37  * named ff7_<type name> and ff8_<type name> respectively.
38  * X - type name
39  * Y - object name
40  * Z - initial value
41  */
42 #define VOBJ(X,Y,Z) struct ff7_ ## X *ff7_ ## Y = ff8 ? 0 : (struct ff7_ ## X *)(Z); struct ff8_ ## X *ff8_ ## Y = ff8 ? (struct ff8_ ## X *)(Z) : 0
43 
44 /*
45  * VPTR - Access the raw pointer contained in an object.
46  * X - object
47  */
48 #define VPTR(X) (ff8 ? (void *)ff8_ ## X : (void *)ff7_ ## X)
49 
50  /*
51  * VPTRCAST - Access the raw pointer contained in an object, casting to a defined type
52  * T - object type
53  * X - object
54  */
55 #define VPTRCAST(T, X) (ff8 ? (struct T *)ff8_ ## X : (struct T *)ff7_ ## X)
56 
57 /*
58  * VASS - Assign a new pointer to an object.
59  * T - object type
60  * X - object
61  * Y - new value
62  */
63 #define VASS(T,X,Y) { if(ff8) ff8_ ## X = (struct ff8_ ## T *)(Y); else ff7_ ## X = (struct ff7_ ## T *)(Y); }
