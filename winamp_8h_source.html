43  void (*Init)(); // called when loaded
44  void (*Quit)(); // called when unloaded
45 
46  int (*Open)(int samplerate, int numchannels, int bitspersamp, int bufferlenms, int prebufferms);
47  // returns >=0 on success, <0 on failure
48 
49  // NOTENOTENOTE: bufferlenms and prebufferms are ignored in most if not all output plug-ins.
50  // ... so don't expect the max latency returned to be what you asked for.
51  // returns max latency in ms (0 for diskwriters, etc)
52  // bufferlenms and prebufferms must be in ms. 0 to use defaults.
53  // prebufferms must be <= bufferlenms
54  // pass bufferlenms==-666 to tell the output plugin that it's clock is going to be used to sync video
55  // out_ds turns off silence-eating when -666 is passed
56 
57  void (*Close)(); // close the ol' output device.
58 
59  int (*Write)(char* buf, int len);
60  // 0 on success. Len == bytes to write (<= 8192 always). buf is straight audio data.
61  // 1 returns not able to write (yet). Non-blocking, always.
62 
63  int (*CanWrite)(); // returns number of bytes possible to write at a given time.
64  // Never will decrease unless you call Write (or Close, heh)
65 
66  int (*IsPlaying)(); // non0 if output is still going or if data in buffers waiting to be
67  // written (i.e. closing while IsPlaying() returns 1 would truncate the song
68 
69  int (*Pause)(int pause); // returns previous pause state
70 
71  void (*SetVolume)(int volume); // volume is 0-255
72  void (*SetPan)(int pan); // pan is -128 to 128
73 
74  void (*Flush)(int t); // flushes buffers and restarts output at time t (in ms)
75  // (used for seeking)
76 
77  int (*GetOutputTime)(); // returns played time in MS
78  int (*GetWrittenTime)(); // returns time written in MS (used for synching up vis stuff)
79 };
